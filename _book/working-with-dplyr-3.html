Chapter 10 Working with dplyr

  • left_join merges two data frames by common columns
  • sample_n shows a few random rows from the data frame
> taxi_cost <- data.frame(game = c("poker", "horse_bets"), 
+                         taxi_cost = c(50, 180), 
+                         stringsAsFactors = F)
> # let's incorporate information on taxi cost
> df_gambling_taxi <- left_join(df_gambling, taxi_cost, by = "game")
